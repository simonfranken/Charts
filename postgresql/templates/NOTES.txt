  $ kubectl run {{ include "postgresql.fullname" . }}-client --rm --image=postgres:16.2 --restart=Never --env=PGPASSWORD=$(kubectl get secret {{ .Values.postgresql.existingSecret }} -o jsonpath="{.data.password}" | base64 -d) --command -- psql --host {{ include "postgresql.fullname" . }} -U postgres

  2. Connect from within your cluster:

  $ psql --host {{ include "postgresql.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local -U postgres
